1. WAP to enter numbers till the user wants. At the end, it should display the count of positive, negative, and zeros entered.
#include <stdio.h>

int main() {
    int num, positive = 0, negative = 0, zero = 0;
    char choice;
    
    do {
        printf("Enter a number: ");
        scanf("%d", &num);
        
        if (num > 0) {
            positive++;
        } else if (num < 0) {
            negative++;
        } else {
            zero++;
        }
        
        printf("Do you want to continue? (y/n): ");
        scanf(" %c", &choice);  // Note the space before %c to consume newline
        
    } while (choice == 'y' || choice == 'Y');
    
    printf("\nResults:\n");
    printf("Positive numbers: %d\n", positive);
    printf("Negative numbers: %d\n", negative);
    printf("Zeros: %d\n", zero);
    
    return 0;
}
Note: The program uses a do-while loop to ensure the user is prompted at least once. The space before %c in scanf is important to handle the newline character from the previous input.

2. WAP to print the multiplication table of the number entered by the user in the format: Num * 1 = Num
#include <stdio.h>

int main() {
    int num, i;
    
    printf("Enter a number to generate its multiplication table: ");
    scanf("%d", &num);
    
    printf("\nMultiplication Table of %d\n", num);
    printf("-----------------------\n");
    
    for (i = 1; i <= 10; i++) {
        printf("%d * %2d = %d\n", num, i, num * i);
    }
    
    return 0;
}
Note: The program uses a for loop to iterate from 1 to 10. The format specifier %2d ensures proper alignment of the numbers in the output.

Sample Output:

Enter a number to generate its multiplication table: 7 Multiplication Table of 7 ----------------------- 7 * 1 = 7 7 * 2 = 14 7 * 3 = 21 ... 7 * 10 = 70
3. WAP to generate the following patterns:

Pattern A:

#include <stdio.h>

int main() {
    int i, j, num = 1, rows;
    
    printf("Enter number of rows: ");
    scanf("%d", &rows);
    
    printf("\nPattern A:\n");
    for (i = 1; i <= rows; i++) {
        for (j = 1; j <= i; j++) {
            printf("%d", num++);
        }
        printf("\n");
    }
    
    // Pattern B
    printf("\nPattern B:\n");
    for (i = 0; i < 5; i++) {
        int current = 1;
        // Print leading spaces
        for (j = 1; j < 5 - i; j++) {
            printf(" ");
        }
        // Print numbers
        for (j = 0; j <= i; j++) {
            printf("%d ", current);
            current = current * (i - j) / (j + 1);
        }
        printf("\n");
    }
    
    return 0;
}
Note: Pattern A uses nested loops to print numbers in a triangular pattern. Pattern B prints Pascal's triangle using the combination formula.

4. The population of a town is 100000. The population has increased steadily at the rate of 10% per year for the last 10 years. Write a program to determine the population at the end of each year in the last decade.
#include <stdio.h>

int main() {
    float population = 100000;
    int year;
    
    printf("Year\tPopulation\n");
    printf("----------------\n");
    
    for (year = 1; year <= 10; year++) {
        population = population + (population * 0.10);
        printf("%2d\t%.0f\n", year, population);
    }
    
    return 0;
}
Note: The program uses a for loop to calculate the population for each year. The population is increased by 10% each year using the formula: population = population + (population * 0.10).

Sample Output:

Year Population ---------------- 1 110000 2 121000 3 133100 ... 10 259374
5. WAP to print all Ramanujan numbers (numbers that can be expressed as the sum of two cubes in two different ways) up to a reasonable limit.
#include <stdio.h>
#include <math.h>

int main() {
    int limit, a, b, c, d, count = 0;
    
    printf("Enter the upper limit to find Ramanujan numbers: ");
    scanf("%d", &limit);
    
    printf("\nRamanujan numbers up to %d:\n", limit);
    printf("-----------------------------\n");
    
    // Check all possible combinations of a, b, c, d
    for (a = 1; a <= cbrt(limit); a++) {
        for (b = a; b <= cbrt(limit); b++) {
            for (c = a + 1; c <= cbrt(limit); c++) {
                for (d = c; d <= cbrt(limit); d++) {
                    if (a != c && a != d && b != c && b != d) {
                        int sum1 = a*a*a + b*b*b;
                        int sum2 = c*c*c + d*d*d;
                        
                        if (sum1 == sum2 && sum1 <= limit) {
                            printf("%d = %d³ + %d³ = %d³ + %d³\n", 
                                   sum1, a, b, c, d);
                            count++;
                        }
                    }
                }
            }
        }
    }
    
    if (count == 0) {
        printf("No Ramanujan numbers found up to %d\n", limit);
    }
    
    return 0;
}
Note: This program uses four nested loops to find all possible combinations of four numbers (a, b, c, d) where a³ + b³ = c³ + d³, which is the definition of a Ramanujan number. The program checks all combinations up to the cube root of the given limit.
